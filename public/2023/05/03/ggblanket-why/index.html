ggblanket-basics

2023-05-03

ggblanket seeks to simplify ggplot2 visualisation.

How?

library(tidyverse)
library(ggblanket)
library(patchwork)
library(palmerpenguins)

penguins <- penguins |>
  mutate(sex = stringr::str_to_sentence(sex)) |>
  tidyr::drop_na(sex)
  1. One-function approach. I.e. No +’s
p1 <- penguins |> #pipe in
  ggplot() + #now plus's
  geom_point(aes(x = flipper_length_mm, y = body_mass_g, col = species))

p2 <- penguins |>
  gg_point(
    x = flipper_length_mm,
    y = body_mass_g,
    col = species
  )


p1 / p2

  1. col and fill as a unified col aesthetic
p1 <- penguins |>
  ggplot() +
  geom_density(aes(x = body_mass_g, fill = species))

p2 <- penguins |>
  gg_density(
    x = body_mass_g, 
    col = species
  )

p1 / p2

  1. unifed approach to changing colours via a pal argument
p1 <- penguins |>
  ggplot() +
  geom_histogram(
    aes(x = body_mass_g),
    fill = "#794044")

p2 <- penguins |>
  gg_histogram(
    x = body_mass_g, 
    pal = "#794044"
  )

p1 / p2

p1 <- penguins |>
  ggplot() +
  geom_jitter(aes(x = species, y = body_mass_g, col = sex)) +
  scale_color_manual(values = c("#fa8072", "#468499"))

p2 <- penguins |>
  gg_jitter(
    x = species, 
    y = body_mass_g, 
    col = sex, 
    pal = c("#fa8072", "#468499"))

p1 / p2

  1. facetting as an aesthetic
p1 <- penguins |>
  ggplot() +
  geom_density(aes(x = flipper_length_mm, fill = sex)) +
  facet_wrap(~species)

p2 <- penguins |>
  gg_density(
    x = flipper_length_mm,
    col = sex,
    facet = species)


p1 / p2

p1 <- penguins |>
  ggplot() +
  geom_histogram(aes(x = flipper_length_mm)) +
  facet_grid(rows = vars(species), cols = vars(sex))

p2 <- penguins |>
  gg_histogram(
    x = flipper_length_mm,
    facet = sex,
    facet2 = species)
